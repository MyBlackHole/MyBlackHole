```shell
CREATE RESOURCE UNIT S1_unit_config
                MEMORY_SIZE = '5G',
                MAX_CPU = 1, MIN_CPU = 1,
                LOG_DISK_SIZE = '6G',
                MAX_IOPS = 10000, MIN_IOPS = 10000, IOPS_WEIGHT=1;

CREATE TENANT IF NOT EXISTS mq_t1 
                PRIMARY_ZONE='zone1',

SELECT * FROM DBA_OB_TENANTS WHERE TENANT_NAME = 'mq_t1';
```
